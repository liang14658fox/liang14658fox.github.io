## 算法1. 将 对象数组 转换为 健值对对象
:::code-group
```js [方案一]
var directionary_2 = [
        {
            value: '1',
            dictlabel: '一级',
        },
        {
            value: '2',
            dictlabel: '二级',
        },
        {
            value: 'aw',
            dictlabel: '三级',
        },
        {
            value: '4',
            dictlabel: '四级',
        }
    ]
let TypeObj ={}
directionary_2.forEach(item=>{
    TypeObj[item.value]=item.dictlabel
})
console.log(TypeObj)

```
```js [方案二]
var directionary_2 = [
        {
            value: '1',
            dictlabel: '一级',
        },
        {
            value: '2',
            dictlabel: '二级',
        },
        {
            value: 'aw',
            dictlabel: '三级',
        },
        {
            value: '4',
            dictlabel: '四级',
        }
    ]
let TypeObj_2={}
directionary_2.forEach(item=>{
    TypeObj_2[item.value]=item[Object.keys(item)[1]]
})
console.log(TypeObj_2)
//得到健值对对象
// {
// 1:'一级',
// 2:'二级',
// 3:'三级',
// 4:'四级'
// }
```
