/**
 * @param {number[]} nums
 * @return {boolean}
 */
var canJump = function (nums) {
    var dp = new Array(nums.length).fill(0);
    dp[0] = 1;
    for (var i = 0; i < nums.length; i++) {
        if(dp[i] === 0)
        return false;
        for (var j = 0; j <= nums[i] && i + j < nums.length; j++) {
            dp[i+j] = 1;
            if(i+j === nums.length-1)
            return true;
        }
    }
    return false;
};
