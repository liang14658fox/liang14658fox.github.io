3. 执行耗时 #

括号内部包含一个标签，计算工作A time 到 timeEnd 的耗时，中间使用 timeLog 可以不间断查看耗时。

js
function doSomething(ms) {
    return new Promise(res => {
        setTimeout(() => res(), ms)
    })
}

console.time('工作A')

await doSomething(150)
console.timeLog('工作A')

await doSomething(250)
console.timeLog('工作A')

console.timeEnd('工作A')
